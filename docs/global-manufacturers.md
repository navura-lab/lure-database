## 国別マッピング用DBスキーマ案

-- manufacturers テーブル拡張案
ALTER TABLE manufacturers ADD COLUMN country_code TEXT; -- ISO 3166-1 alpha-2
ALTER TABLE manufacturers ADD COLUMN country_name_ja TEXT; -- 日本語国名
ALTER TABLE manufacturers ADD COLUMN region TEXT; -- 地域: asia, north-america, europe, oceania, etc.
ALTER TABLE manufacturers ADD COLUMN founded_year INTEGER;
ALTER TABLE manufacturers ADD COLUMN headquarters TEXT;
ALTER TABLE manufacturers ADD COLUMN parent_company TEXT;
